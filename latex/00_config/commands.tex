% Theorem styles

\newtheorem{lemma}[theorem]{Lemma}
\newtheorem{proposition}[theorem]{Proposition}
\newtheorem{corollary}[theorem]{Corollary}

\theoremstyle{definition}
\newtheorem{definition}[theorem]{Definition}

\theoremstyle{remark}
\newtheorem{remark}[theorem]{Remark}
\newtheorem{example}[theorem]{Example}


% Epigraph settings
\setlength{\epigraphwidth}{0.5\textwidth}
\setlength{\epigraphrule}{0pt}
\epigraphfontsize{\small}
% vertical gap between epigraph text and author
\setlength{\epigraphsourceoffset}{2em}
\renewcommand{\textflush}{flushright} % default
\renewcommand{\epigraphsource}[1]{\vspace{5em}\hfill\textit{#1}}

% Page style and numbering
\pagestyle{fancy}
\fancyhf{} % clear
\fancyfoot[C]{\thepage} % centered page number
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}

% Chapter title style

  {\normalfont\LARGE\bfseries}
  {Chapter \thechapter}
  {2ex}
  {\LARGE\bfseries}
\titlespacing*{\chapter}{5pt}{-1em}{2.5em}

  \renewcommand{\thechapter}{\Alph{chapter}}%
}
\makeatother
